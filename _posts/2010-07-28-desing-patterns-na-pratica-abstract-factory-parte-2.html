Continuando o artigo sobre as Fábricas Abstratas, iremos executar um teste prático utilizando C#.

Exemplo Prático

Nosso exemplo prático constitui uma concessionária de veículos novos e semi-novos. Nossa concessionária, trabalha com duas montadoras: Suzuki e Honda. Cada montadora produz dois tipos de produtos: Moto e Carro. Em nossa concessionária, iremos vender um produto para cada tipo(Carro e Moto), de ambas as montadoras.

Segue o diagrama do exemplo proposto:

As fábricas concretas do diagrama são representadas pelas montadoras (Suzuki e Honda). Ainda que as montadoras criem Produtos diferentes (Titan, Fit, Katana e Vitara), as interações entre as montadoras e os produtos são os mesmos, pois eles são definidos pelos tipos (ProdutosAbstratos A e B - Carros e Motos).

   1: namespace Abstract_Factory
   2: {
   3:     class Program
   4:     {
   5:         static void Main(string[] args)
   6:         {
   7:             // Crio as fábricas concretas que serão utilizadas
   8:             FabricaHonda honda = new FabricaHonda();
   9:             FabricaSuzuki suzuki = new FabricaSuzuki();
  10:
  11:             // Crio os automóveis usando as fábricas correspondentes
  12:             Automoveis automovel = new Automoveis(honda);
  13:             automovel = new Automoveis(suzuki);
  14:         }
  15:     }
  16:
  17: #region AbstractFactory ------------------------------------------------------------------
  18:
  19:     abstract class FabricaMontadora
  20:     {
  21:         // Métodos para fabricação dos Produtos       
  22:         public abstract Moto CriaMoto();
  23:         public abstract Carro CriaCarro();
  24:     }
  25:
  26: #endregion
  27:
  28: #region ConcreteFactory ------------------------------------------------------------------
  29:
  30:     /// <summary>
  31:     /// Fábrica Concreta 1
  32:     /// </summary>
  33:     class FabricaHonda : FabricaMontadora
  34:     {
  35:         public override Moto CriaMoto() {return new Titan();}
  36:
  37:         public override Carro CriaCarro() {return new Fit();}
  38:     }
  39:
  40:     /// <summary>
  41:     /// Fábrica Concreta 2
  42:     /// </summary>
  43:     class FabricaSuzuki : FabricaMontadora
  44:     {
  45:         public override Moto CriaMoto() {return new Katana();}
  46:
  47:         public override Carro CriaCarro() {return new Vitara();}
  48:     }
  49:
  50: #endregion
  51:
  52: #region AbstractProduct ------------------------------------------------------------------
  53:
  54:     /// <summary>
  55:     /// Produto Abstrato A
  56:     /// </summary>
  57:     abstract class Moto {}
  58:
  59:     /// <summary>
  60:     /// Produto Abstrato B
  61:     /// </summary>
  62:     abstract class Carro {}
  63:
  64: #endregion
  65:
  66: #region ConcreteProduct ------------------------------------------------------------------
  67:
  68:     /// <summary>
  69:     /// Define um objeto (produto) para ser criado pela Fábrica Concreta correspondente 
  70:     /// Produto A1
  71:     /// </summary>
  72:     class Titan : Moto {}
  73:
  74:     /// <summary>
  75:     /// Produto B1
  76:     /// </summary>
  77:     class Fit : Carro {}
  78:
  79:     /// <summary>
  80:     /// Produto A2
  81:     /// </summary>
  82:     class Katana : Moto {}
  83:
  84:     /// <summary>
  85:     /// Produto A3
  86:     /// </summary>
  87:     class Vitara : Carro {}
  88:
  89: #endregion
  90:
  91: #region Client ---------------------------------------------------------------------------
  92:
  93:     /// <summary>
  94:     /// Usa apenas interfaces declaradas pela AbstractFactory e AbstractProduct
  95:     /// </summary>
  96:     class Automoveis
  97:     {
  98:         private Moto moto;
  99:         private Carro carro;
 100:
 101:         public Automoveis(FabricaMontadora fabrica)
 102:         {
 103:             moto = fabrica.CriaMoto();
 104:             carro = fabrica.CriaCarro();
 105:         }
 106:     }
 107:
 108: #endregion
 109: }
